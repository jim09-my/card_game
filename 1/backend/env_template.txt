# 后端服务环境变量配置示例
# 如需使用，复制此文件为 .env 并修改配置（仅在生产环境需要）

# ============================================
# 数据库配置
# ============================================
# 默认使用SQLite（开发环境，无需配置）
# DATABASE_URL=sqlite:///./card_game.db

# 使用PostgreSQL（生产环境）
# DATABASE_URL=postgresql://用户名:密码@localhost:5432/card_game_db

# 使用MySQL（生产环境）
# DATABASE_URL=mysql+pymysql://用户名:密码@localhost:3306/card_game_db

# ============================================
# JWT配置（可选，有默认值）
# ============================================
# 生成随机密钥：openssl rand -hex 32
# JWT_SECRET_KEY=你的随机密钥字符串

# JWT算法（默认HS256）
# ALGORITHM=HS256

# Token过期时间（分钟，默认30）
# ACCESS_TOKEN_EXPIRE_MINUTES=30

# ============================================
# 服务器配置（可选）
# ============================================
# HOST=0.0.0.0
# PORT=8000
